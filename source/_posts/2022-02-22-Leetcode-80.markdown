写麻烦了，总之就是双指针，一个快指针一个慢指针

```java
class Solution {
   
public:
    int removeDuplicates(vector<int>& nums) {
   
        int len=nums.size();
        int cnt=1,L=1;
        for(int i=1;i<len;i++){
   
            if(nums[i]==nums[i-1]){
   
                if(cnt==2) continue;
                else cnt++,L++;
            }
            else{
   
                L++;
                cnt=1;
            }
        }
        // cout<<L<<endl;
        cnt=1;
        for(int i=1,j=1;i<len && j<L;i++){
   
            if(nums[i]==nums[i-1] && cnt==2) continue;
            if(nums[i]!=nums[i-1]) cnt=0;
            nums[j++]=nums[i];
            cnt++;
        }
        return L;
    }
};
```
